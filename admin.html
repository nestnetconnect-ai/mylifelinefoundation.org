🔗 Google Sheets Setup — Step by Step Guide

Ye ek baar ka kaam hai. Iske baad Volunteers, Donors aur Content sab automatically Google Sheet mein save hoga. Completely Free!

1
Google Sheet banao

sheets.new pe jao → Ek naya sheet khulega → Name do: MLF Website Data

2
Apps Script open karo

Sheet mein: upar Extensions menu click karo → Apps Script click karo → pura code delete karo

3
Ye code copy karke Apps Script mein paste karo: 
function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var d = JSON.parse(e.postData.contents);
    var tab = ss.getSheetByName(d.tab) || ss.insertSheet(d.tab);
    if (d.action === 'append') {
      tab.appendRow(d.row);
    } else if (d.action === 'write') {
      tab.clearContents();
      if (d.rows && d.rows.length)
        tab.getRange(1,1,d.rows.length,d.rows[0].length).setValues(d.rows);
    }
    return ContentService.createTextOutput(
      JSON.stringify({status:'ok'})
    ).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({status:'error',msg:err.toString()})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var tab = ss.getSheetByName(e.parameter.tab);
    if (!tab) return ContentService.createTextOutput(
      JSON.stringify({data:[]})
    ).setMimeType(ContentService.MimeType.JSON);
    return ContentService.createTextOutput(
      JSON.stringify({data: tab.getDataRange().getValues()})
    ).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({data:[]})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
